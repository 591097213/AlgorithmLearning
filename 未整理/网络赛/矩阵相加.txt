#include<stdio.h>//¡¾¼ÇµÃÇåÁã¡¿
void main()
{
	int a[100][100],b[100][100],c[100][100],i,j,m1,n1,m2,n2,m,n,k;
	while(scanf("%d%d%d%d",&n1,&m1,&n2,&m2)!=EOF)
	{
		n=m1>n2?m1:n2;
		m=n1>m2?n1:m2;
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				c[i][j]=a[i][j]=b[i][j]=0;
			}
		}//C=A=B=0
		for(i=0;i<n1;i++)
		{
			for(j=0;j<m1;j++)
			{
				scanf("%d",&a[j][i]);
			}
		}//AT
		for(i=0;i<n2;i++)
		{
			for(j=0;j<m2;j++)
			{
				scanf("%d",&b[i][j]);
			}
		}//B
		for(i=0;i<n;i++)
		{
			k=m-1;
			for(j=0;j<m;j++)
			{
				
				c[i][j]=a[i][j]+b[i][j];
				printf("%d",c[i][j]);
				if(k>0)
				{printf(" ");k--;}
			}
			printf("\n");
		}
	}
}
