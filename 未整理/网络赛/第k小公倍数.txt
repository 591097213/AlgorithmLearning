#include<stdio.h>
void main()
{
	int a,b,i,j,k;
	while(scanf("%d%d%d",&a,&b,&k)!=EOF)
	{
		j=0;
		for(i=a>b?a:b;;i++)
		{
			if(i%a==0&&i%b==0)
				j++;
			if(j==k)
				break;
		}
		printf("%d\n",i);
	}
}
