#include<stdio.h>
#include<math.h>
void main()
{
	int a[100][102],m,n,i,j,s,man,min;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&m);
			s=0;
			for(j=0;j<m;j++)
			{
				scanf("%d",&a[i][j]);
				s+=a[i][j];
			}
			a[i][100]=s/m;
		}
		scanf("%d",&man);//input²¢¼ÆËã
		for(i=0;i<n;i++)
		{
			a[i][101]=abs(a[i][100]-man);
		}
		min=a[0][101];
		for(i=0;i<n;i++)
		{
			if(a[i][101]<min)
				min=a[i][101];
		}//Çómin
		for(i=n-1;i>=0;i--)
		{
			if(min==a[i][101])
				break;
		}
		printf("%d\n",i+1);
	}
}
