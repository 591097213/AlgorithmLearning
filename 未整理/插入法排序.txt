#include<stdio.h>//input n ¼°n¸öÊý
void main()
{
	int a[100],i,j,key,n;
	while(~scanf("%d",&n))
	{
		for(i=0;i<n;i++)
		{scanf("%d",&a[i]);}
		for(i=1;i<n;i++)
		{
			key=a[i];
			j=i-1;
			while(a[j]<key&&j>=0)
			{
				a[j+1]=a[j];
				j--;
			}
			a[j+1]=key;
		}
		for(i=0;i<n;i++)
		{printf("%d ",a[i]);}
		printf("\n");
	}
}
