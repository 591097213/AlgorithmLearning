/*问题描述
　　如果一个序列满足下面的性质，我们就将它称为摆动序列：
　　1. 序列中的所有数都是不大于k的正整数；
　　2. 序列中至少有两个数。
　　3. 序列中的数两两不相等；
　　4. 如果第i C 1个数比第i C 2个数大，则第i个数比第i C 2个数小；如果第i C 1个数比第i C 2个数小，则第i个数比第i C 2个数大。
　　比如，当k = 3时，有下面几个这样的序列：
　　1 2
　　1 3
　　2 1
　　2 1 3
　　2 3
　　2 3 1
　　3 1
　　3 2
　　一共有8种，给定k，请求出满足上面要求的序列的个数。
输入格式
　　输入包含了一个整数k。（k<=20）
输出格式
　　输出一个整数，表示满足要求的序列个数。
样例输入
3
样例输出
8*/


/**************************************
            解法1
*************************************/
//递归求解，从第一个位置开始每次递归按顺序试一个数，该数满足条件则解++，>n则结束本次递归
#include<iostream>
using namespace std;

int a[25];//存放数列
int n;//
int ans;//结果
int visit[25];//是否放置过

int IsCan(int num,int k)//是否可放置，k是放置的位置，num是要放置的数
{
	if(k==1 || k==2) return 1;//第一第二个位置无条件可放
	if(a[k-1]>a[k-2] && num<a[k-2]) return 1;
	if(a[k-1]<a[k-2] && num>a[k-2]) return 1;
	return 0;//其他情况，不可放置
}

void dfs(int cur)
{
	if(cur>n) return ;
	for(int i=1;i<=n;i++)//从1到n分别放置在第一个位置
	{
		if(!visit[i] && IsCan(i,cur))//未访问过且可放置
		{
			visit[i]=1;//置1
			a[cur]=i;
			if(cur>=2)//不一定要把n个整数都放置完，只要是>=2就认为是新数列
			{
				ans++;
				/*for(int i=1;i<=cur;i++){
					cout<<a[i]<<" ";
				}
				cout<<endl;
				*/
			}
			dfs(cur+1);
			visit[i]=0;//调用返回时候顺便置零
		}
	} 
}

int main()
{
	cin>>n;
	dfs(1);//从第一个位置开始放置
	cout<<ans<<endl;
}

//从k个数中任取i个，
//这几个数若要满足条件4)，只能让这些数中数值大小位于中间的两个数放排头，
//那么只可能有两种排列（一次先升后减另一次先减后升）(只要第一个数和第二个数的大小确定,排列必然唯一)
//本质上是计算2(C(K,2)+C(K,3)....+C(K,K)）,
#include <stdio.h>
#define ll long long//64位整数
int fac(short i)//最大10!=3628800
{
    short t = 1;
    int s = 1;
    for(;t <= i;t++)
        s *= t;
    return s;
}

ll combinate(short i,short k)//求组合数C(k,i)
{
    short t;
    ll s = 1;
    if(i > k/2)
        i = k - i;
    for(t=0;t<i;t++)
        s *= (k-t);
    return s/fac(i);
}

int main()
{
    short k,i;
    ll s = 0;//累加结果
    scanf("%hd",&k);//输入k
    for(i=2;i<=k;i++)
        s += combinate(i,k);
    printf("%d\n",2*s);
    return 0;
}
